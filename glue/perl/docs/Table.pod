=head1 NAME

Apache::*::Table - Table classes for Apache::Request, Apache::Upload, and Apache::Cookie.

=for testing
    use Apache2;
    use Apache::Upload;
    use Apache::Cookie;
    use APR::Pool;




=head1 SYNOPSIS

=for example begin

    my $table = Apache::Request::Table->new(APR::Pool->new);
    $table->{test} = 1;
    $table->{foo} = "bar1";
    $table->add(foo => "bar2");

    {
        my $do_data = "";
        $table->do( sub { $do_data .= "@_"; 1 } );
        ok $do_data eq "test 1foo bar1foo bar2";
    }

=for example end
